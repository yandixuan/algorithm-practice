# 最长回文子串

## 中心扩展

```java
class Solution {
    public String longestPalindrome(String s) {
        if (s == null || s.length() < 1) {
            return "";
        }
        int start = 0, end = 0;
        for (int i = 0; i < s.length(); i++) {
            int len1 = expandAroundCenter(s, i, i);
            int len2 = expandAroundCenter(s, i, i + 1);
            int len = Math.max(len1, len2);
            if (len > end - start) {
                start = i - (len - 1) / 2;
                end = i + len / 2;
            }
        }
        return s.substring(start, end + 1);
    }

    private int expandAroundCenter(String s, int left, int right) {
        int l = left, r = right;
        while (l >= 0 && r < s.length() && s.charAt(l) == s.charAt(r)) {
            l--;
            r++;
        }
        return r - l - 1;
    }
}
```

```java
    public String longestPalindrome(String s) {
        // 把字符串处理成奇数位
        StringBuilder t = new StringBuilder("#");
        for (int i = 0; i < s.length(); ++i) {
            t.append(s.charAt(i));
            t.append('#');
        }
        t.append('#');
        s = t.toString();

        // 该数组记录原字符每一个字符对应的最长回文半径
        int[] p = new int[s.length()];
        // 最长回文中心右边界、中心索引
        int right = -1, mid = -1;
        for (int i = 0; i < s.length(); ++i) {
            // 确定一个最小的半径
            int r = 1;
            // 如果i在边界内
            if (right >= i) {

                /**
                 * i关于中心的对称点mid k坐标为 2 * mid - i
                 * 如果 p[2 * mid - i]的半径 + i 超过 right 即  p[2 * mid - i]> right-i 则 i点回文半径为 right-i
                 * 由此可将表达式优化成 取 Math.min(right - i, p[2 * mid - i])的最小值
                 */
                r = Math.min(right - i, p[2 * mid - i]);
            }
            // 中心扩展搜索
            while (i - r >= 0 && i + r < s.length() && s.charAt(i - r) == s.charAt(i + r)) {
                r++;
            }
            // 如果 i+r-1 对应的边界超过了 right 即更新 最大回文右边界及中心点
            if (i + r - 1 > right) {
                right = i + r - 1;
                mid = i;
            }
            p[i] = r;
        }
        // 最大半径
        int maxLength = 0;
        int centerIndex = 0;
        // 遍历数组p取找到最大回文半径
        for (int i = 0; i < p.length; i++) {
            if (p[i] > maxLength) {
                maxLength = p[i];
                centerIndex = i;
            }
        }
        StringBuilder sb = new StringBuilder();
        // 起点
        int start = centerIndex - maxLength + 1;
        // 终点(不包含)
        int end = centerIndex + maxLength;
        for (int i = start; i < end; i++) {
            if (s.charAt(i) == '#') {
                continue;
            }
            sb.append(s.charAt(i));
        }
        return sb.toString();
    }
```
